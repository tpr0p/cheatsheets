\textbf{Dynamics}
\begin{align}
  L &= T - V\\
  \frac{d}{dt}(\frac{\partial L}{\partial \dot{q}^i}) &= \frac{\partial L}{\partial q^i}\\
  p_i &= \frac{\partial L}{\partial \dot{q}^i}\\
  H &= \sum_{i = 1}^{N} p_i \dot{q}^i - L\\
  \frac{dq^i}{dt} &= \frac{\partial H}{\partial p_i}\\
  \frac{dp_i}{dt} &= -\frac{\partial H}{\partial q^i}
\end{align}

\textbf{Kinematics}
\begin{align}
  v_f &= v_i + a t\\
  x_f &= x_i + v_it+ \frac{1}{2}at^2\\
  x_f &= x_i + \frac{1}{2}(v_i + v_f)t\\
  v_f^2 &= v_i^2 + 2 a (x_f - x_i)
\end{align}
