\textbf{TLS}
  \begin{align}
    H/\hbar &= -\Omega/2 \ket{0} \bra{1} -\Omega^{*}/2 \ket{1} \bra{0} - \Delta \ket{1} \bra{1}\\
    \Omega &= \frac{d_{01} E}{\hbar}; \quad \Delta = \nu - \omega_{10}\\
    \rho_{01} &= \frac{i\Omega}{2(\gamma/2 - i\Delta)} \frac{1}{1 + s}\\
    s &= \frac{\lvert \Omega \rvert^2}{2(\gamma^2/4 + \Delta^2)}
  \end{align}

  \textbf{AMO}
  \begin{align}
    \langle \vec{F} \rangle / \hbar &= \frac{1}{2} \textrm{Re}[\frac{\rho_{01}}{\Omega}]\vec{\nabla} \lvert \Omega \rvert^2
    + \textrm{Im}[\frac{\rho_{01}}{\Omega}] \lvert \Omega \rvert^2 \vec{\nabla} \phi\\
    \langle \vec{F} \rangle_{\textrm{dipole}} / \hbar &\approx \frac{-1}{4 \Delta} \vec{\nabla} \lvert \Omega \rvert^2\\
    \langle \vec{F} \rangle_{\textrm{dis}} / \hbar &= -\vec{k} \frac{\lvert \Omega \rvert^2 \gamma}{\gamma^2 + 4 \Delta^2} \frac{1}{1 + s}
  \end{align}
