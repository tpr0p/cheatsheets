\begin{align}

  f(t) e^{-i \omega t} dt\\

  F(k_x, k_y) &= (2\pi)^{-2} \int_{\mathbb{R}^2} f(x, y)
  e^{i(k_x x + k_y y)} dx dy\\
  f(x, y) &= \int_{\mathbb{R}^2} F(k_x, k_y)
  e^{-i(k_x x + k_y y)} dk_x dk_y\\

  \mathcal{F}^{-1}[G * F] &= f(x, y) g(x, y)\\
\end{align}
