\noindent
\textbf{Wigner-Seitz Unit Cell}
\begin{enumerate}
  \item Choose one lattice point. Draw lines to every other lattice point.
  \item Bisect each line with a plane normal to the line.
  \item Space bounded by planes forms the unit cell.
\end{enumerate}
